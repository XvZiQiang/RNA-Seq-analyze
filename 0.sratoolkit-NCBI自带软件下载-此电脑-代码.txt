# 下载 sra-tools 安装包
cd ~
wget https://ftp-trace.ncbi.nlm.nih.gov/sra/sdk/current/sratoolkit.current-ubuntu64.tar.gz

# 解压
tar zxvf sratoolkit.current-ubuntu64.tar.gz

# 自动检测解压后的文件夹名（一般为 sratoolkit.3.2.1-ubuntu64）
TOOLKIT_DIR=$(find ~ -maxdepth 1 -type d -name "sratoolkit.*-ubuntu64" | head -n 1)

# 加入环境变量
echo "export PATH=\$PATH:$TOOLKIT_DIR/bin" >> ~/.bashrc
source ~/.bashrc

# 给整个 bin 目录添加执行权限
chmod +x $TOOLKIT_DIR/bin/*

# 验证 prefetch 是否安装成功
prefetch -V
